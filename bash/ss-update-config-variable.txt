####################################################################################################
#### C. SS-Update-Config-Variable: Message (Begin Script) #############################################
####################################################################################################

## this is a simple message that announces to the shell the purpose of this bash script ##
## it will only be seen by sudo users who manually run this script in the shell ##

ss_echo "${COLOR_INFO}Running ss-update-config-variable... ${COLOR_RESET}"

####################################################################################################
#### D. SS-Update-Config-Variable: Validate Arguments ##############################################
####################################################################################################

## check that the required arguments are provided ##
if [ "$#" -ne 2 ]; then
    ss_echo "${COLOR_ERROR}Usage: $0 <var_name> <var_value>${COLOR_RESET}"
    exit 1
fi

VAR_NAME="$1"
VAR_VALUE="$2"

## validate that the variable exists in ss-config ##
if ! grep -q "^${VAR_NAME}=" "${PATH_SS_CONFIG}"; then
    ss_echo "${COLOR_ERROR}Variable ${VAR_NAME} not found in ss-config!${COLOR_RESET}"
    exit 1
fi

## define a whitelist of allowed variables to update ##
WHITELIST=(
    "SS_REMOTE_BACKUPS"
    "RCLONE_TYPE"
    "RCLONE_REMOTE_PATH"
    "RCLONE_CLIENT_ID"
    "RCLONE_CLIENT_SECRET"
    "RCLONE_USER"
    "RCLONE_PASSWORD"
    "RCLONE_TOKEN"
    "RCLONE_BACKUP_PATH"
    "RCLONE_PARALLEL_TRANSFERS"
    "RSYNC_REMOTE_HOST"
    "RSYNC_BACKUP_PATH"
    "RSYNC_USER"
    "RSYNC_PASSWORD"
)

## check if the variable is in the whitelist ##
if [[ ! " ${WHITELIST[@]} " =~ " ${VAR_NAME} " ]]; then
    ss_echo "${COLOR_ERROR}Variable ${VAR_NAME} is not allowed for updating.${COLOR_RESET}"
    exit 1
fi

####################################################################################################
#### E. SS-Update-Config-Variable: Update Variable in SS-Config ####################################
####################################################################################################

## here we update the variable's value in ss-config ##

ss_sed "s|^${VAR_NAME}=.*|${VAR_NAME}=\"${VAR_VALUE}\"|g" "${PATH_SS_CONFIG}"
## sudo echo "${SFTP_USER}:${SFTP_PASSWORD_NEW}" | sudo /usr/sbin/chpasswd ##

####################################################################################################
#### F. SS-Update-Config-Variable: Message (Display Update Message) ################################
####################################################################################################

ss_echo ""
ss_echo "${COLOR_INFO}Updated ${VAR_NAME} in ss-config:${COLOR_RESET}"
ss_echo "${COLOR_WARN}NEW ${VAR_NAME} Value: ${VAR_VALUE} ${COLOR_RESET}"
ss_echo ""
